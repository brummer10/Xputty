37 void main_run(Xputty *main) {
38  Widget_t * wid = main->childlist->childs[0];
39  Atom WM_DELETE_WINDOW;
40  WM_DELETE_WINDOW = XInternAtom(wid->app->dpy, "WM_DELETE_WINDOW", True);
41  XSetWMProtocols(wid->app->dpy, wid->widget, &WM_DELETE_WINDOW, 1);
42 
43  XEvent xev;
44  int ew;
45 
46  while (main->run && (XNextEvent(main->dpy, &xev)>=0)) {
47  ew = childlist_find_widget(main->childlist, xev.xany.window);
48  if(ew >= 0) {
49  Widget_t * w = main->childlist->childs[ew];
50  w->event_callback(w, &xev, main, NULL);
51  }
52 
53  switch (xev.type) {
54  case ButtonPress:
55  if(main->hold_grab != NULL) {
56  Widget_t *view_port = main->hold_grab->childlist->childs[0];
57  bool is_item = False;
58  int i = view_port->childlist->elem-1;
59  for(;i>-1;i--) {
60  Widget_t *w = view_port->childlist->childs[i];
61  if (xev.xbutton.window == w->widget) {
62  is_item = True;
63  break;
64  }
65  }
66  if (xev.xbutton.window == view_port->widget) is_item = True;
67  if (!is_item) {
68  XUngrabPointer(main->dpy,CurrentTime);
69  widget_hide(main->hold_grab);
70  main->hold_grab = NULL;
71  }
72  }
73  break;
74  case ClientMessage:
75  /* delete window event */
76  if (xev.xclient.data.l[0] == WM_DELETE_WINDOW &&
77  xev.xclient.window == wid->widget) {
78  main->run = false;
79  } else {
80  int i = childlist_find_widget(main->childlist, xev.xclient.window);
81  if(i>=1) destroy_widget(main->childlist->childs[i],main);
82  }
83  break;
84  }
85  }
86 }

main_init
void main_init(Xputty *main)
main_init - open the Display and init the main->childlist. Set the bool run to true. The bool run is used to terminate the main event loop. main_init() should be called directly after the declaration of Xputty before the first Widget_t get created. Any Widget_t created afterwards will be added to the main childlist. The main childlist is used to check if a Widget_t is valid to receive a Event. Xputty check if a Widget_t is registerd in the main childlist, and only forward events when it found the Widget_t in the list. When a Widget_t call destroy_widget() any childs of this Widget_t receive a call to destroy_widget() to release there memory, they get removed from the main childlist and finaly the Widget_t itself will be removed from the main childlist as well. On main_quit() any remaining Widget_t from the main childlist will be destroyed, to ensure that we leave the memory clean.
