110 Colors *get_color_scheme(Xputty *main, Color_state st) {
111  switch(st) {
112  case NORMAL_:
113  return &main->color_scheme->normal;
114  break;
115  case PRELIGHT_:
116  return &main->color_scheme->prelight;
117  break;
118  case SELECTED_:
119  return &main->color_scheme->selected;
120  break;
121  case ACTIVE_:
122  return &main->color_scheme->active;
123  break;
124  default:
125  return &main->color_scheme->normal;
126  break;
127  }
128  return NULL;

201 void set_pattern(Widget_t *w, Colors *from, Colors *to, Color_mod mod) {
202  double *col_from = NULL;
203  double *col_to = NULL;
204  switch (mod) {
205  case FORGROUND_:
206  col_from = from->fg;
207  col_to = to->fg;
208  break;
209  case BACKGROUND_:
210  col_from = from->bg;
211  col_to = to->bg;
212  break;
213  case BASE_:
214  col_from = from->base;
215  col_to = to->base;
216  break;
217  case TEXT_:
218  col_from = from->text;
219  col_to = to->text;
220  break;
221  case SHADOW_:
222  col_from = from->shadow;
223  col_to = to->shadow;
224  break;
225  case FRAME_:
226  col_from = from->frame;
227  col_to = to->frame;
228  break;
229  case LIGHT_:
230  col_from = from->light;
231  col_to = to->light;
232  break;
233  }
234  XWindowAttributes attrs;
235  XGetWindowAttributes(w->app->dpy, (Window)w->widget, &attrs);
236  int width = attrs.width;
237  int height = attrs.height;
238  cairo_pattern_t *pat = cairo_pattern_create_linear (0, 0, width, height);
239  cairo_pattern_add_color_stop_rgba(pat, 0.0, col_from[0], col_from[1], col_from[2], col_from[3]);
240  cairo_pattern_add_color_stop_rgba(pat, 1.0, col_to[0], col_to[1], col_to[2], col_to[3]);
241  cairo_set_source(w->crb, pat);
242  cairo_set_source(w->cr, pat);
243  cairo_pattern_destroy (pat);
244 }
