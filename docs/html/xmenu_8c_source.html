26 void pop_menu_show(Widget_t *parent, Widget_t *menu, int elem, bool above) {
27  if (!childlist_has_child(menu->childlist)) return;
28  Widget_t* view_port = menu->childlist->childs[0];
29  if (!view_port->childlist->elem) return;
30  _configure_menu(parent, menu, elem, above);
31  pop_widget_show_all(menu);
32  int err = XGrabPointer(menu->app->dpy, DefaultRootWindow(parent->app->dpy), True,
33  ButtonPressMask|ButtonReleaseMask|PointerMotionMask,
34  GrabModeAsync, GrabModeAsync, None, None, CurrentTime);
35  menu->app->hold_grab = menu;
36 
37  if (err) debug_print("Error grap pointer\n");
38 }

51 Widget_t* create_menu(Widget_t *parent, int height) {
52 
53  int x1, y1;
54  Window child;
55  XTranslateCoordinates( parent->app->dpy, parent->widget, DefaultRootWindow(parent->app->dpy), 0, 0, &x1, &y1, &child );
56  Widget_t *wid = create_window(parent->app, DefaultRootWindow(parent->app->dpy), x1, y1, 10, height);
57  create_viewport(wid, 10, 5*height);
58  Atom window_type = XInternAtom(wid->app->dpy, "_NET_WM_WINDOW_TYPE", False);
59  long vale = XInternAtom(wid->app->dpy, "_NET_WM_WINDOW_TYPE_POPUP_MENU", False);
60  XChangeProperty(wid->app->dpy, wid->widget, window_type,
61  XA_ATOM, 32, PropModeReplace, (unsigned char *) &vale,1 );
62  XSetTransientForHint(parent->app->dpy,wid->widget,parent->widget);
63  wid->func.expose_callback = _draw_menu;
64  wid->flags |= IS_POPUP;
65  wid->scale.gravity = NONE;
66  childlist_add_child(parent->childlist, wid);
67  return wid;
68 }

70 Widget_t* menu_add_item(Widget_t *menu,const char * label) {
71  Widget_t* view_port = menu->childlist->childs[0];
72  XWindowAttributes attrs;
73  XGetWindowAttributes(menu->app->dpy, (Window)menu->widget, &attrs);
74  int width = attrs.width;
75  int height = attrs.height;
76  int si = childlist_has_child(view_port->childlist);
77  Widget_t *wid = create_widget(menu->app, view_port, 0, height*si, width, height);
78  float max_value = view_port->adj->max_value+1.0;
79  set_adjustment(view_port->adj,0.0, 0.0, 0.0, max_value,1.0, CL_VIEWPORT);
80  wid->scale.gravity = MENUITEM;
81  wid->flags &= ~USE_TRANSPARENCY;
82  wid->label = label;
83  wid->func.expose_callback = _draw_item;
84  wid->func.enter_callback = transparent_draw;
85  wid->func.leave_callback = transparent_draw;
86  return wid;
87 }
