38

40  cairo_set_line_width(w->crb, 1.0);
41  _pattern_out(w, NORMAL_, height);
42  cairo_fill_preserve(w->crb);
43  use_frame_color_scheme(w, PRELIGHT_);
44  } else if(w->state==1) {
45  _pattern_out(w, PRELIGHT_, height);

51  cairo_fill_preserve(w->crb);

55  _pattern_in(w, ACTIVE_, height);
56  cairo_fill_preserve(w->crb);
57  cairo_set_line_width(w->crb, 1.0);
58  use_frame_color_scheme(w, PRELIGHT_);
59  }
60  cairo_stroke(w->crb);
61 
62  if(w->state==2) {
63  cairo_rectangle(w->crb,4.0, 6.0, width, height);
64  cairo_stroke(w->crb);
65  cairo_rectangle(w->crb,3.0, 4.0, width, height);
66  cairo_stroke(w->crb);
67  } else if (w->state==3) {
68  cairo_rectangle(w->crb,3.0, 4.0, width, height);

70  }
71 
72  float offset = 0.0;

95  cairo_line_to(w->crb, wa+offset, h+offset);
96  cairo_fill(w->crb);

98 }

102  if (!w) return;
103  XWindowAttributes attrs;
104  XGetWindowAttributes(w->app->dpy, (Window)w->widget, &attrs);

110

127  cairo_stroke(w->crb);
128 
129  cairo_rectangle(w->crb,4.0, 4.0, width, height);
130  cairo_stroke(w->crb);
131  cairo_rectangle(w->crb,3.0, 3.0, width, height);
132  cairo_stroke(w->crb);
133 
134  cairo_text_extents_t extents;
135  if(w->state==0) {
136  use_fg_color_scheme(w, NORMAL_);
137  } else if(w->state==1) {
138  use_fg_color_scheme(w, PRELIGHT_);
139  } else if(w->state==2) {
140  use_fg_color_scheme(w, SELECTED_);
141  }
142 
143  use_text_color_scheme(w, get_color_state(w));
144  cairo_set_font_size (w->crb, height/2.3);
145  cairo_select_font_face (w->crb, "Sans", CAIRO_FONT_SLANT_NORMAL,
146  CAIRO_FONT_WEIGHT_BOLD);
147  cairo_text_extents(w->crb,w->label , &extents);
148 
149  cairo_move_to (w->crb, (width-extents.width)*0.5, (height+extents.height)*0.5);
150  cairo_show_text(w->crb, w->label);

152  if (extents.width > (float)width-20) {
153  tooltip_set_text(w,w->label);
154  w->flags |= HAS_TOOLTIP;
155  } else {
156  w->flags &= ~HAS_TOOLTIP;
157  }
