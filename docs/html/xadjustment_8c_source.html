83 void adj_set_motion_state(void *w, float x, float y) {
84  Widget_t * wid = (Widget_t*)w;
85  const float scaling = 1.0;
86  if(wid->adj_x) {
87  float value= wid->adj_x->value;
88  switch(wid->adj_x->type) {
89  case (CL_CONTINUOS):
90  {
91  float state = (wid->adj_x->start_value - wid->adj_x->min_value) /
92  (wid->adj_x->max_value - wid->adj_x->min_value);
93  float nsteps = wid->adj_x->step / (wid->adj_x->max_value - wid->adj_x->min_value);
94  float nvalue = min(1.0,max(0.0,state + ((float)(x - wid->pos_x)*scaling *nsteps)));
95  value = nvalue * (wid->adj_x->max_value - wid->adj_x->min_value) + wid->adj_x->min_value;
96  }
97  break;
98  case (CL_TOGGLE):
99  // dont toggle on motion!
100  // value = wid->adj_x->value ? 0.0 : 1.0;
101  break;
102  default:
103  break;
104  }
105  check_value_changed(wid->adj_x, &value);
106  }
107  if(wid->adj_y) {
108  float value = wid->adj_y->value;
109  switch(wid->adj_y->type) {
110  case (CL_CONTINUOS):
111  {
112  float state = (wid->adj_y->start_value - wid->adj_y->min_value) /
113  (wid->adj_y->max_value - wid->adj_y->min_value);
114  float nsteps = wid->adj_y->step / (wid->adj_y->max_value - wid->adj_y->min_value);
115  float nvalue = min(1.0,max(0.0,state + ((float)(wid->pos_y - y)*scaling *nsteps)));
116  value = nvalue * (wid->adj_y->max_value - wid->adj_y->min_value) + wid->adj_y->min_value;
117  }
118  break;

122  break;
123  default:
124  break;
125  }
126  check_value_changed(wid->adj_y, &value);
127  }
128 }

133  adj->value = *(value);
