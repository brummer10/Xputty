40 static void draw_window(void *w_, void* user_data) {
41  Widget_t *w = (Widget_t*)w_;
42  XWindowAttributes attrs;
43  XGetWindowAttributes(w->app->dpy, (Window)w->widget, &attrs);
44  int width = attrs.width;
45  int height = attrs.height;
46  if (attrs.map_state != IsViewable) return;
47 
48  cairo_rectangle(w->crb,0,0,width,height);
49  set_pattern(w,&w->app->color_scheme->selected,&w->app->color_scheme->normal,BACKGROUND_);
50  cairo_fill (w->crb);
51 
52  widget_set_scale(w);
53  use_fg_color_scheme(w, NORMAL_);
54  cairo_set_font_size (w->crb, 12.0);
55  cairo_select_font_face (w->crb, "Sans", CAIRO_FONT_SLANT_NORMAL,
56  CAIRO_FONT_WEIGHT_BOLD);
57 
58  cairo_move_to (w->crb, 20, 35);
59  cairo_show_text(w->crb, "Directory");
60  cairo_move_to (w->crb, 20, 85);
61  cairo_show_text(w->crb, "File");
62  cairo_move_to (w->crb, 20, 340);
63  cairo_show_text(w->crb, "Load: ");
64  cairo_move_to (w->crb, 45, 380);
65  cairo_show_text(w->crb, "Show hidden files");
66  cairo_move_to (w->crb, 60, 340);
67  cairo_show_text(w->crb, w->label);
68  widget_reset_scale(w);
69 }

254 Widget_t *open_file_dialog(Widget_t *w, const char *path, const char *filter) {
255  FileDialog *file_dialog = (FileDialog*)malloc(sizeof(FileDialog));
256 
257  file_dialog->fp = (FilePicker*)malloc(sizeof(FilePicker));
258  fp_init(file_dialog->fp, (path) ? path : "/");
259  file_dialog->parent = w;
260  file_dialog->send_clear_func = true;
261  file_dialog->icon = NULL;
262 
263  file_dialog->w = create_window(w->app, DefaultRootWindow(w->app->dpy), 0, 0, 660, 420);

265  file_dialog->w->parent_struct = file_dialog;
266  widget_set_title(file_dialog->w, "File Selector");

269  widget_set_icon_from_png(file_dialog->w,file_dialog->icon,LDVAR(directory_png));
270 
271  file_dialog->ct = add_combobox(file_dialog->w, "", 20, 40, 580, 30);
272  file_dialog->ct->parent_struct = file_dialog;
273  file_dialog->ct->func.value_changed_callback = combo_response;
