40 static void draw_message_window(void *w_, void* user_data) {

54  int height = cairo_xlib_surface_get_height(w->image);
55  double x = 64.0/(double)(width);
56  double y = 64.0/(double)height;
57  double x1 = (double)height/64.0;
58  double y1 = (double)(width)/64.0;

60  cairo_set_source_surface (w->crb, w->image, 50, 50);
61  cairo_rectangle(w->crb,50, 50, width, height);
62  cairo_fill(w->crb);
63  cairo_scale(w->crb, x1,y1);
64 
65  draw_message_label(w,width_t,height_t);
66  widget_reset_scale(w);
67 }

156 static void message_okay_callback(void *w_, void* user_data) {
157  Widget_t *w = (Widget_t*)w_;
158  if (w->flags & HAS_POINTER && !*(int*)user_data){
159  Widget_t *p = (Widget_t*)w->parent;
160  MessageBox *mb = p->parent_struct;
161  if(mb->message_type == QUESTION_BOX || mb->message_type == SELECTION_BOX) {
162  Widget_t *pa = (Widget_t*)p->parent;
163  pa->func.dialog_callback(pa,&mb->response);
164  } else if(mb->message_type == ENTRY_BOX) {
165  Widget_t *pa = (Widget_t*)p->parent;
166  if (strlen( mb->text_entry->input_label))
167  mb->text_entry->input_label[strlen( mb->text_entry->input_label)-1] = 0;
168  mb->text_entry->label = mb->text_entry->input_label;
169  pa->func.dialog_callback(pa,&mb->text_entry->label);
170  }
171  destroy_widget(p, p->app);
172  }

222 static void entry_get_text(void *w_, void *key_, void *user_data) {
223  Widget_t *w = (Widget_t*)w_;
224  if (!w) return;
225  XKeyEvent *key = (XKeyEvent*)key_;
226  if (!key) return;
227  int nk = key_mapping(w->app->dpy, key);
228  if (nk) {
229  switch (nk) {
230  case 10:
231  {
232  Widget_t *p = (Widget_t*)w->parent;
233  MessageBox *mb = p->parent_struct;
234  Widget_t *pa = (Widget_t*)p->parent;
235  if (strlen( mb->text_entry->input_label))
236  mb->text_entry->input_label[strlen( mb->text_entry->input_label)-1] = 0;
237  mb->text_entry->label = mb->text_entry->input_label;
238  pa->func.dialog_callback(pa,&mb->text_entry->label);
239 
240  destroy_widget(p, p->app);
241  }
242  break;
243  case 11: entry_clip(w);
244  break;
245  default:
246  break;
247  }
248  } else {
249  Status status;
250  KeySym keysym;
251  char buf[32];
252  Xutf8LookupString(w->xic, key, buf, sizeof(buf) - 1, &keysym, &status);
253  if(status == XLookupChars || status == XLookupBoth){
254  entry_add_text(w, buf);
255  }
256  }
257 }

270 static void check_for_message(MessageBox *mb, const char *message) {
271  if(!message) return;
272  if(!strlen(message)) return;
273  int len = 0;
274  char *ms =strdup(message);
275  char * p = strtok (ms, "|");
276  while (p) {
277  mb->message = realloc (mb->message, sizeof (char*) * ++mb->lin);
278  mb->message[mb->lin-1] = strdup(p);
279  len = max(len, strlen(mb->message[mb->lin-1]));
280  p = strtok (NULL, "|");
281  }
282  free(ms);
283  mb->width = len*12 ;
284  mb->height = mb->lin*12+100;
285 }

330 Widget_t *open_message_dialog(Widget_t *w, int style, const char *title,
331  const char *message, const char *choices) {
332 
333  MessageBox *mb = (MessageBox*)malloc(sizeof(MessageBox));
334  mb->response = 0;
335  mb->message_type = 0;
336  mb->lin = 0;
337  mb->width = 330;
338  mb->height = 200;
339  mb->message = NULL;
340  mb->sel = 0;
341  mb->choices = NULL;
342  mb->icon = NULL;
343  check_for_message(mb, message);
344  check_for_choices(mb, choices);
345  check_for_style(mb, style);
346  Widget_t *wid = create_window(w->app, DefaultRootWindow(w->app->dpy), 0, 0, mb->width, mb->height);
347  wid->label = message;
348  wid->flags |= HAS_MEM;
349  wid->scale.gravity = CENTER;
350  wid->parent_struct = mb;
351  wid->parent = w;
352  wid->func.mem_free_callback = mg_mem_free;
353  wid->func.expose_callback = draw_message_window;
354  char *alternate_title = NULL;
355  char *button_title = "OK";
356  switch (style) {
357  case INFO_BOX:
358  widget_get_png(wid, LDVAR(info_png));
359  alternate_title = "INFO";
360  mb->message_type = INFO_BOX;
361  widget_set_icon_from_surface(wid,mb->icon,wid->image);
362  break;
363  case WARNING_BOX:
364  widget_get_png(wid, LDVAR(warning_png));
365  alternate_title = "WARNING";
366  mb->message_type = WARNING_BOX;
367  widget_set_icon_from_surface(wid,mb->icon,wid->image);
368  break;
369  case ERROR_BOX:
370  widget_get_png(wid, LDVAR(error_png));
371  alternate_title = "ERROR";
372  mb->message_type = ERROR_BOX;
373  widget_set_icon_from_surface(wid,mb->icon,wid->image);
374  break;
375  case QUESTION_BOX:
376  widget_get_png(wid, LDVAR(question_png));
377  alternate_title = "QUESTION";
378  Widget_t *no = add_button(wid, "NO", 10, mb->height-40, 60, 30);
379  no->scale.gravity = CENTER;
380  no->func.value_changed_callback = message_no_callback;
381  mb->message_type = QUESTION_BOX;
382  button_title = "YES";
383  widget_set_icon_from_surface(wid,mb->icon,wid->image);
384  break;
385  case SELECTION_BOX:
386  widget_get_png(wid, LDVAR(choice_png));
387  alternate_title = "SELECTION";
388  mb->message_type = SELECTION_BOX;
389  create_checkboxes(wid);
390  widget_set_icon_from_surface(wid,mb->icon,wid->image);
391  break;
392  case ENTRY_BOX:
393  widget_get_png(wid, LDVAR(message_png));
394  alternate_title = "TEXT ENTRY";
395  mb->message_type = ENTRY_BOX;
396  create_entry_box(wid);
397  widget_set_icon_from_surface(wid,mb->icon,wid->image);
398  break;
399  }
400  widget_set_title(wid, (title)? title : alternate_title);
401 
402  Widget_t *okay = add_button(wid, button_title, mb->width-70, mb->height-40, 60, 30);
403  okay->scale.gravity = CENTER;
404  okay->func.value_changed_callback = message_okay_callback;
405 
406  widget_show_all(wid);
407  return wid;
408 }
