24 void _scroll_event(Widget_t * wid, int direction) {
25  Adjustment_t *adj = NULL;
26  if (wid->adj_y) {
27  adj = wid->adj_y;
28  } else if(wid->adj_x) {
29  adj = wid->adj_x;
30  }
31  if (adj) {
32  float value = adj->value;
33  switch(adj->type) {
34  case (CL_CONTINUOS):
35  value = min(adj->max_value,max(adj->min_value,
36  adj->value + (adj->step * direction)));
37  break;
38  case (CL_VIEWPORT):
39  case (CL_ENUM):
40  value = min(adj->max_value,max(adj->min_value,
41  adj->value + (adj->step * -direction)));
42  break;
43  case (CL_TOGGLE):
44  // value = adj->value ? 1.0 : 0.0;
45  break;
46  default:
47  break;
48  }
49  check_value_changed(adj, &value);
50  }
51 }

165  switch (nk) {
166  case 3: _set_adj_value(wid, false, 1*n);

168  case 4: _set_adj_value(wid, true, 1*n);
169  break;
170  case 5: _set_adj_value(wid, false, -1*n);
171  break;
172  case 6: _set_adj_value(wid, true, -1*n);
173  break;
174  case 10:
175  {
176  int i = 0;
177  for(;i<wid->childlist->elem;i++) {
178  Widget_t *w = wid->childlist->childs[i];

182  }
183  }

204 void _has_pointer(Widget_t *w, XButtonEvent *button) {
205  XWindowAttributes attrs;

207

209  (button->x>0 && button->y>0)){

250 void _resize_childs(Widget_t *wid) {
251  if(!childlist_has_child(wid->childlist)) return;
252  int i = 0;
253  for(;i<wid->childlist->elem;i++) {
254  Widget_t *w = wid->childlist->childs[i];
255  switch(w->scale.gravity) {
256  case(NORTHWEST):
257  XResizeWindow (wid->app->dpy, w->widget, max(1,
258  w->scale.init_width - (wid->scale.scale_x)),
259  max(1,w->scale.init_height - (wid->scale.scale_y)));
260  break;
261  case(NORTHEAST):
262  XResizeWindow (wid->app->dpy, w->widget, max(1,
263  w->scale.init_width - (wid->scale.scale_x)), w->height);
264  break;
265  case(SOUTHWEST):
266  XMoveWindow(wid->app->dpy,w->widget,w->scale.init_x-wid->scale.scale_x,
267  w->scale.init_y-wid->scale.scale_y);
268 
269  break;
270  case(SOUTHEAST):
271  XMoveWindow(wid->app->dpy,w->widget,w->scale.init_x,
272  w->scale.init_y-wid->scale.scale_y);
273  break;
274  case(CENTER):
275  XMoveWindow(wid->app->dpy,w->widget,w->scale.init_x /
276  wid->scale.cscale_x,w->scale.init_y / wid->scale.cscale_y);
277  XResizeWindow (wid->app->dpy, w->widget, max(1,
278  w->scale.init_width / (wid->scale.cscale_x)),
279  max(1,w->scale.init_height / (wid->scale.cscale_y)));
280  break;
281  case(ASPECT):
282  XMoveWindow(wid->app->dpy,w->widget,w->scale.init_x /
283  wid->scale.ascale,w->scale.init_y / wid->scale.ascale);
284  XResizeWindow (wid->app->dpy, w->widget, max(1,
285  w->scale.init_width / (wid->scale.ascale)),
286  max(1,w->scale.init_height / (wid->scale.ascale)));
287 
288  break;
289  case(MENUITEM):
290  XResizeWindow (wid->app->dpy, w->widget, max(1,
291  w->scale.init_width - (wid->scale.scale_x)-5), w->scale.init_height);
292  break;
293  case(NONE):
294  break;
295  default:
296  break;
297  }
298  w->func.configure_notify_callback(w,NULL);
299  }
